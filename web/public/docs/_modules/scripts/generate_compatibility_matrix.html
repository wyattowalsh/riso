    def parse_copier_yml(self) -> None:
        """Parse copier.yml and extract prompt definitions."""
        with self.copier_yml_path.open() as f:
            config = yaml.safe_load(f)

        self.prompts = config.get("prompts", {})
        self.defaults = config.get("defaults", {})

        # Extract dependencies from 'when' clauses
        for prompt_name, prompt_config in self.prompts.items():
            if isinstance(prompt_config, dict) and "when" in prompt_config:
                deps = self._extract_dependencies(prompt_config["when"])
                if deps:
                    self.dependencies[prompt_name] = deps

            # Extract default dependencies
            if isinstance(prompt_config, dict) and "default" in prompt_config:
                default_deps = self._extract_dependencies(str(prompt_config["default"]))
                if default_deps:
                    existing = self.dependencies.get(prompt_name, [])
                    self.dependencies[prompt_name] = list(set(existing + default_deps))



    def _extract_dependencies(self, jinja_expr: str) -> list[str]:
        """Extract variable names from Jinja2 expressions.

        Args:
            jinja_expr: Jinja2 template expression

        Returns:
            List of variable names referenced in expression
        """
        # Match variable names in Jinja2 expressions
        # Handles: {{ var }}, {{ 'value' in var }}, {{ var == 'value' }}
        matches = re.findall(r'\b([a-z_][a-z0-9_]*)\b', jinja_expr)
        # Filter out common Jinja2/Python keywords
        keywords = {'if', 'else', 'elif', 'in', 'and', 'or', 'not', 'true', 'false', 'none'}
        return [m for m in matches if m not in keywords]

    def _analyze_incompatibilities(self) -> None:
        """Analyze prompts to identify known incompatibilities."""
        # Define known incompatible combinations
        self.incompatibilities = {
            "frontend_framework": [
                ("none", "component_library", "Component libraries require a frontend framework"),
                ("none", "css_framework", "CSS frameworks require a frontend framework"),

                ("[]", "rust_web_framework", "Rust framework requires Rust in backend_languages"),
                ("[]", "node_api_framework", "Node framework requires Node in backend_languages"),
            ],
        }

    def generate_matrix_table(self) -> str:
        """Generate markdown compatibility matrix table.

        Returns:
            Markdown table string
        """
        table = ["## Compatibility Matrix", "", "### Backend Language Compatibility", ""]

        # Backend frameworks compatibility
        table.extend([
            "| Language | Frameworks | Package Managers | Testing |",
            "|----------|------------|------------------|---------|",
        ])

        backend_compat = {
            "Python": ("FastAPI, Flask, Django", "uv, poetry, conda", "pytest, unittest"),
            "Rust": ("Axum, Actix, Rocket, Warp, Poem", "cargo", "cargo test"),
            "Node.js": ("Fastify, Express, NestJS, Koa, Hono", "pnpm, npm, yarn", "Jest, Vitest"),
            "Go": ("Gin, Fiber, Echo, Chi", "go modules", "go test"),
            "Java": ("Spring Boot, Quarkus, Micronaut", "Maven, Gradle", "JUnit, TestNG"),
        }

        for lang, (frameworks, pkg_mgrs, testing) in backend_compat.items():
            table.append(f"| {lang} | {frameworks} | {pkg_mgrs} | {testing} |")

        table.extend(["", "### Frontend Framework Compatibility", ""])

        table.extend([
            "| Framework | Meta-Frameworks | State Management | Routing | Component Libraries |",
            "|-----------|-----------------|------------------|---------|---------------------|",
        ])

        frontend_compat = {
            "React": (
                "Vite, Next.js, Remix, Gatsby",
                "Zustand, Redux, MobX, Jotai",
                "TanStack Router, React Router",
                "shadcn/ui, MUI, Chakra UI, Mantine",
            ),
            "Vue": (
                "Vite, Nuxt, Quasar, VitePress",
                "Pinia, Vuex",
                "Vue Router",
                "Element Plus, Vuetify, Quasar",
            ),
            "Svelte": (
                "SvelteKit, Vite",
                "Svelte stores",
                "SvelteKit routing",
                "Skeleton UI, Carbon Components",
            ),
        }

        for framework, (meta, state, routing, components) in frontend_compat.items():
            table.append(f"| {framework} | {meta} | {state} | {routing} | {components} |")

        return "\n".join(table)

    def generate_alternatives(self) -> str:
        """Generate alternative recommendations section.

        alternatives = ["## Alternative Recommendations", ""]

        recommendations = {
            "Backend Alternatives": {

                "Rust": "Consider Go for easier learning curve with good performance",
                "Node.js": "Consider Bun for even faster execution (experimental)",

                "Next.js": "Consider Remix for web standards focus, Astro for content sites",

                "Vite": "Consider Turbopack (Next.js), esbuild for maximum speed",

        for category, alts in recommendations.items():

    def generate_documentation(self) -> str:
        """Generate complete compatibility matrix documentation.

        Returns:
            Complete markdown documentation
        """
        self._analyze_incompatibilities()

        sections = [
            "# Riso Template Compatibility Matrix",
            "",
            "This document provides a comprehensive compatibility reference for all Riso template options.",
            "Use this guide to understand which combinations work together and make informed decisions.",
            "",
            "<!-- AUTO-GENERATED: Do not edit manually. Run `uv run python scripts/generate_compatibility_matrix.py` -->",
            "",
            self.generate_matrix_table(),
            "",
            "## Decision Trees",
            "",
            "Use these decision trees to guide your configuration choices:",
            "",
            *self.generate_decision_trees(),
            "",
            self.generate_alternatives(),
            "",
            self.generate_best_practices(),
            "",
            "## Incompatible Combinations",
            "",
            "The following combinations are **not supported** and will fail validation:",
            "",
        ]

        # Add incompatibility details
        for prompt, incompats in self.incompatibilities.items():
            sections.append(f"### {prompt}")
            sections.append("")
            for value, related_prompt, reason in incompats:
                sections.append(f"- `{prompt}={value}` + `{related_prompt}` → {reason}")
            sections.append("")

        sections.extend([
            "## Prompt Dependencies",
            "",
            "The following prompts depend on other prompt values:",
            "",
        ])

        for prompt, deps in sorted(self.dependencies.items()):
            deps_str = ", ".join(f"`{d}`" for d in deps)
            sections.append(f"- **{prompt}** depends on: {deps_str}")

        return "\n".join(sections)

    def visualize_dependencies(self) -> str:
        """Generate Mermaid diagram of prompt dependencies.

        Returns:
            Mermaid diagram showing all prompt relationships
        """
        lines = ["```mermaid", "graph LR"]

        # Add nodes for major prompt categories
        categories = {
            "backend": ["backend_languages", "python_api_framework", "rust_web_framework", "node_api_framework"],
            "frontend": ["frontend_framework", "react_framework", "vue_framework", "component_library"],
            "structure": ["project_layout", "repository_structure", "monorepo_tool"],
            "docs": ["docs_site", "docs_theme_mode", "docs_search_provider"],
        }

        for category, prompts in categories.items():
            lines.append(f"    subgraph {category.capitalize()}")
            for prompt in prompts:
                if prompt in self.prompts:
                    lines.append(f"        {prompt}")
            lines.append("    end")

        # Add dependency edges
        for prompt, deps in self.dependencies.items():
            for dep in deps:
                if dep in self.prompts:
                    lines.append(f"    {dep} --> {prompt}")

        lines.append("```")
        return "\n".join(lines)

def main() -> int:
    """Main entry point."""
    parser = argparse.ArgumentParser(
        description="Generate compatibility matrix from copier.yml"
    )
    parser.add_argument(
        "--copier-yml",
        type=Path,
        default=Path("template/copier.yml"),
        help="Path to copier.yml (default: template/copier.yml)",
    )
    parser.add_argument(
        "--output",
        type=Path,
        default=Path("docs/compatibility-matrix.md"),
        help="Output path (default: docs/compatibility-matrix.md)",
    )
    parser.add_argument(
        "--visualize-deps",
        action="store_true",
        help="Generate dependency visualization diagram",
    )
    args = parser.parse_args()

    if not args.copier_yml.exists():
        print(f"Error: {args.copier_yml} not found", file=sys.stderr)
        return 1

    generator = CompatibilityMatrixGenerator(args.copier_yml)
    generator.parse_copier_yml()
    generator.save_documentation(args.output)

    if args.visualize_deps:
        viz_path = args.output.parent / "compatibility-dependencies.md"
        viz = generator.visualize_dependencies()
        viz_path.write_text(f"# Prompt Dependencies\n\n{viz}\n")
        print(f"✓ Dependency visualization: {viz_path}")

    return 0




if __name__ == "__main__":
    sys.exit(main())
